    <script>
        let rowCount = 12;

        const updateTotals = () => {
            let total = 0;
            document.querySelectorAll('.amount').forEach(input => {
                total += parseFloat(input.value) || 0;
            });
            document.getElementById('grand-total').innerText = total.toFixed(2);
        };

        document.querySelectorAll('.amount').forEach(input => {
            input.addEventListener('input', updateTotals);
        });

        const addRow = () => {
            rowCount++;
            const tableBody = document.querySelector('#subscriptions-table tbody');
            const newRow = document.createElement('tr');
            newRow.innerHTML = `
                <td><input type="text" id="service-${rowCount}" placeholder="Service"></td>
                <td><input type="date" id="date-${rowCount}"></td>
                <td><input type="number" class="amount" id="amount-${rowCount}"></td>
            `;
            tableBody.insertBefore(newRow, document.querySelector('.totals'));
            document.querySelectorAll('.amount').forEach(input => {
                input.addEventListener('input', updateTotals);
            });
            saveToLocalStorage();
        };

        document.getElementById('add-row-button').addEventListener('click', () => {
            addRow();
            const button = document.getElementById('add-row-button');
            button.classList.add('active');
            setTimeout(() => {
                button.classList.remove('active');
            }, 300);
        });

        const saveToLocalStorage = () => {
            const inputs = document.querySelectorAll('input');
            inputs.forEach(input => {
                localStorage.setItem(input.id, input.value);
            });
        };

        const loadFromLocalStorage = () => {
            const inputs = document.querySelectorAll('input');
            inputs.forEach(input => {
                if (localStorage.getItem(input.id)) {
                    input.value = localStorage.getItem(input.id);
                }
            });
        };

        document.addEventListener('DOMContentLoaded', () => {
            loadFromLocalStorage();
            document.querySelectorAll('input').forEach(input => {
                input.addEventListener('input', () => {
                    saveToLocalStorage();
                    updateTotals();
                });
            });
            updateTotals();
        });

        const themeButton = document.getElementById('theme-button');
        const container = document.querySelector('.container');
        let isDarkTheme = false;

        themeButton.addEventListener('click', () => {
            themeButton.classList.add('active');
            setTimeout(() => {
                themeButton.classList.remove('active');
            }, 300);

            isDarkTheme = !isDarkTheme;
            if (isDarkTheme) {
                container.classList.add('dark-theme');
            } else {
                container.classList.remove('dark-theme');
            }
        });
    </script>
